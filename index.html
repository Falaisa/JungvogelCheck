document.addEventListener("DOMContentLoaded", function () { const plzInput = document.getElementById("plz"); if (plzInput) { plzInput.addEventListener("change", pruefePLZ); } }); async function pruefePLZ() { const plz = document.getElementById("plz").value; let sichtungen = await fetch('sichtungen.json').then(res => res.json()); if (sichtungen.some(sichtung => sichtung.plz === plz)) { alert("Es gibt bereits eine Sichtung in dieser PLZ!"); } else { alert("Keine Sichtungen gefunden."); } }
